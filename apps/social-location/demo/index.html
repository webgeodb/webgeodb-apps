📍 更新用户位置

// 更新用户位置并设置隐私级别 await locationService.updateLocation('user1', { latitude: 39.9042, longitude: 116.4074 }, { accuracy: 10, privacyLevel: 'neighborhood' // 街区级别 }); // 查询附近用户 const nearby = await locationService.findNearbyUsers( { latitude: 39.9042, longitude: 116.4074 }, 5000, // 5km 半径 { limit: 20 } );

🔒 隐私保护设置

// 设置隐私级别 await privacyService.setPrivacyLevel('user1', { level: 'city', // 城市级别 precision: 3000, // 3km 精度 exceptions: ['friend1', 'friend2'] // 例外好友 }); // 设置隐私区域（如家、公司） await privacyService.setPrivacyZones('user1', [{ name: '家', location: { lat: 39.9042, lng: 116.4074 }, radius: 200, // 200米 privacyLevel: 'region' // 区域级别 }]);

🤖 智能推荐

// 基于位置推荐地点 const recommendations = await recommendationService.recommendPlaces({ userId: 'user1', location: { latitude: 39.9042, longitude: 116.4074 }, limit: 10 }); // 推荐附近好友 const friends = await recommendationService.recommendFriends( 'user1', { latitude: 39.9042, longitude: 116.4074 }, 5 ); // 更新用户偏好 await recommendationService.updateUserPreferences('user1', { categories: ['restaurant', 'cafe', 'park'], favoritePlaces: ['place1', 'place2'] });

// 创建新地点 const place = await placeService.createPlace({ name: '故宫博物院', location: { latitude: 39.9163, longitude: 116.3971 }, category: 'museum', rating: 4.8, description: '中国明清两代的皇家宫殿', createdBy: 'user1' }); // 搜索附近地点 const nearby = await placeService.searchNearby( { latitude: 39.9042, longitude: 116.4074 }, 5000, { limit: 10, categories: ['museum'], minRating: 4.0 } );
